# BIG-PHISH Makefile

.PHONY: help install install-dev clean test run docker-build docker-up docker-down lint format setup service

# Variables
PYTHON = python3
VENV = venv
VENV_BIN = $(VENV)/bin
PIP = $(VENV_BIN)/pip
PYTEST = $(VENV_BIN)/pytest
BLACK = $(VENV_BIN)/black
FLAKE8 = $(VENV_BIN)/flake8

help:
	@echo "🐋 BIG-PHISH Makefile Commands:"
	@echo "  make install     - Install dependencies"
	@echo "  make install-dev - Install dev dependencies"
	@echo "  make clean       - Clean temporary files"
	@echo "  make test        - Run tests"
	@echo "  make run         - Run BIG-PHISH"
	@echo "  make docker-build - Build Docker image"
	@echo "  make docker-up   - Start Docker containers"
	@echo "  make docker-down - Stop Docker containers"
	@echo "  make lint        - Run linters"
	@echo "  make format      - Format code"
	@echo "  make setup       - Full setup"
	@echo "  make service     - Install systemd service"
	@echo "  make dashboard   - Run metrics dashboard"

install:
	$(PYTHON) -m venv $(VENV)
	$(PIP) install --upgrade pip setuptools wheel
	$(PIP) install -r requirements.txt

install-dev:
	$(PYTHON) -m venv $(VENV)
	$(PIP) install --upgrade pip setuptools wheel
	$(PIP) install -r requirements.txt
	$(PIP) install -r requirements-full.txt

clean:
	rm -rf $(VENV)
	rm -rf __pycache__
	rm -rf .pytest_cache
	rm -rf .coverage
	rm -rf htmlcov
	rm -rf .mypy_cache
	rm -rf .bigphish/*.db
	rm -rf logs/*.log
	find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	find . -type f -name "*.pyc" -delete

test:
	$(PYTEST) tests/ -v --cov=. --cov-report=html

run:
	source $(VENV)/bin/activate && $(PYTHON) bigphish.py

docker-build:
	docker build -t bigphish:latest .
	docker tag bigphish:latest bigphish:$(shell git rev-parse --short HEAD)

docker-up:
	docker-compose up -d

docker-down:
	docker-compose down

docker-logs:
	docker-compose logs -f

lint:
	$(FLAKE8) . --max-line-length=120 --ignore=E501,W503
	$(PYTHON) -m mypy --ignore-missing-imports .

format:
	$(BLACK) --line-length 120 .

setup: install-dev
	./setup.sh

service:
	sudo cp bigphish.service /etc/systemd/system/
	sudo systemctl daemon-reload
	sudo systemctl enable bigphish
	sudo systemctl start bigphish

dashboard:
	source $(VENV)/bin/activate && $(PYTHON) metrics_dashboard.py

watch:
	source $(VENV)/bin/activate && $(PYTHON) bigphish.py --watch

security:
	source $(VENV)/bin/activate && bandit -r . -f json -o bandit-report.json

coverage:
	source $(VENV)/bin/activate && $(PYTEST) --cov=. --cov-report=html --cov-report=term

all: clean install-dev test lint
